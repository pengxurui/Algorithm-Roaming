## [5. 最长回文子串](https://leetcode.cn/problems/longest-palindromic-substring/)

## 题目描述

给你一个字符串 s，找到 s 中最长的回文子串。

## 题目考点

- 思路 1 - 动态规划：最优子结构：f(i,j) = f(i-1,j+1) and s[i] == s[j]：
- 思路 2 - 中心扩展：枚举所有的回文中心（长度为1或2的子串）并尝试扩展，直到无法扩展为止，此时的回文串长度即为此回文中心下的最长回文串长度，对所有的长度求出最大值
- 思路 3 - Manacher 算法：Hard

## 题解一（动态规划）
 
```
class Solution {
    fun longestPalindrome(s: String): String {
        // 思路 1 - 动态规划：最优子结构：f(i,j) = f(i-1,j+1) & s[i] == s[j]

        val size = s.length

        if (size <= 1) {
            return s
        }

        // 如果没有长度大于 1 的回文串，返回 s[0] 亦是回文串
        var from = 0
        var maxLength = 1

        // 子问题记录
        val dp = Array<BooleanArray>(size) { BooleanArray(size) { false } }

        // 基准情况：长度为 1 的串都是子串
        for (index in 0 until size) {
            dp[index][index] = true
        }

        // 遍历所有长度的子串
        for (L in 2..size) {
            // 长度为 L 的滑动窗口
            for (i in 0 until size) {
                val j = L + i - 1
                if (j >= size) {
                    // 右边界越界
                    break
                }
                // i = 0,j = 2
                if (s[i] != s[j]) {
                    // 子串左右两端不同
                    dp[i][j] = false
                } else {
                    if (L > 3) {
                        // 取决于子问题
                        dp[i][j] = dp[i + 1][j - 1]
                    } else {
                        dp[i][j] = true
                    }
                }
                // 发现新的回文串
                if (dp[i][j]) {
                    if (j - i + 1 > maxLength) {
                        maxLength = j - i + 1
                        from = i
                    }
                }
            }
        }
        return s.substring(from, from + maxLength)
    }
}

class Solution {
    fun longestPalindrome(s: String): String {
        val n = s.length
        val dp = Array(n) { BooleanArray(n) }.apply {
            // 长度为 1 的子串都是回文串
            for (index in 0 until n) {
                this[index][index] = true
            }
        }
        var maxLength = 1
        var startIndex = 0
        for (window in 2..n) {
            for (left in 0 until n) {
                val right = window + left - 1
                if (right >= n) break
                if (s[left] != s[right]) {
                    dp[left][right] = false
                } else {
                    if (window <= 2) {
                        dp[left][right] = true
                    } else {
                        dp[left][right] = dp[left + 1][right - 1]
                    }
                }
                if (dp[left][right] && window > maxLength) {
                    maxLength = window
                    startIndex = left
                }
            }
        }
        return s.substring(startIndex, startIndex + maxLength)
    }
}
```

**复杂度分析：**

- 时间复杂度：O(n^2)
- 空间复杂度：O(n^2) 

## 题解二

```
class Solution {
    fun longestPalindrome(s: String): String {
        if (s.length < 2) {
            return s
        }

        fun expandAndFind(leftVal: Int, rightVal: Int): Int {
            var left = leftVal
            var right = rightVal
            while (left >= 0 && right < s.length && s[left] == s[right]) {
                left--
                right++
            }
            // right - left + 1 - 2 退出循环时多计算了一次
            return right - left - 1
        }

        var maxLength = 1
        var from = 0

        for (index in s.indices) {
            // 长度为 1 的中心
            val len1 = expandAndFind(index, index)
            // 长度为 2 的中心
            val len2 = expandAndFind(index, index + 1)
            val curMaxLength = Math.max(len1, len2)
            if (curMaxLength > maxLength) {
                // 1 - (2 - 1) / 2
                from = index - (curMaxLength - 1) / 2
                maxLength = curMaxLength
            }
        }

        return s.substring(from, from + maxLength)
    }
}
```

**复杂度分析：**

- 时间复杂度：O(n^2)
- 空间复杂度：O(1)
