
- 通用输入输出：

```
import java.util.*
import java.io.*
import java.math.*

object Helper {
    private var token = StringTokenizer("")
    private val input = BufferedReader(InputStreamReader(System.`in`))
    private val output = BufferedWriter(OutputStreamWriter(System.`out`))

    fun close() {
        input.close()
        output.flush()
        output.close()
    }

    fun nextLine() = input.readLine()

    fun next(): String {
        while (!token.hasMoreTokens()) {
            token = StringTokenizer(input.readLine())
        }
        return token.nextToken()
    }

    fun nextInt(): Int = next().toInt()

    fun nextLong(): Long = next().toLong()

    fun nextShort(): Short = next().toShort()

    fun nextByte(): Byte = next().toByte()
    
    fun nextChar(): Char = next()[0]

    fun nextDouble(): Double = next().toDouble()

    fun nextFloat(): Float = next().toFloat()

    fun nextBigInteger(): BigInteger = next().toBigInteger()

    fun nextIntArray(n: Int): IntArray = IntArray(n) { nextInt() }

    fun nextLongArray(n: Int): LongArray = LongArray(n) { nextLong() }

    fun nextFloatArray(n: Int): FloatArray = FloatArray(n) { nextFloat() }

    fun println(a: Int) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Int) {
        output.write("$a")
    }

    fun println(a: Long) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Long) {
        output.write("$a")
    }

    fun println(a: Short) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Short) {
        output.write("$a")
    }

    fun println(a: Char) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Char) {
        output.write("$a")
    }

    fun println(a: Byte) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Byte) {
        output.write("$a")
    }

    fun println(a: Float) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Float) {
        output.write("$a")
    }

    fun println(a: Double) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: Double) {
        output.write("$a")
    }

    fun println(a: BigDecimal) {
        output.write("$a")
        output.newLine()
    }

    fun print(a: BigDecimal) {
        output.write("$a")
    }

    fun println(a: String) {
        output.write(a)
        output.newLine()
    }

    fun print(a: String) {
        output.write(a)
    }
}
```
